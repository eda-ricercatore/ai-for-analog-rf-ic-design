	\frametitle{Project Information}
	\framesubtitle{Design -- High Level Proposal (2-3 sentences + Image)}
	
	\begin{itemize}
	\item Use semiconductor device models of memristors and floating-gate MOSFETs to approximate memristive and floating-gate MOSFET -based analog in-memory computing systems
	\item Implement analog in-memory computing systems by replacing SRAM cell in SRAM subsystem with approximate RAM cell circuit model, based on the aforementioned semiconductor device models
	\item Use recursive self-improvement to design the analog in-memory computing systems, where hardware generators are created to train the AI model (from memory subsystem with memristors or floating-gate MOSFETs to analog in-memory computing systems)
	\end{itemize}
	
	
\end{frame}
