\documentclass[tikz]{standalone}
\usepackage{pgfgantt}
\title{Gantt Charts with the pgfgantt Package}
\begin{document}

%
% A fairly complicated example from section 2.9 of the package
% documentation. This reproduces an example from Wikipedia:
% http://en.wikipedia.org/wiki/Gantt_chart
%
\definecolor{barblue}{RGB}{153,204,254}
\definecolor{groupblue}{RGB}{51,102,254}
\definecolor{linkred}{RGB}{165,0,33}
\renewcommand\sfdefault{phv}
\renewcommand\mddefault{mc}
\renewcommand\bfdefault{bc}
\setganttlinklabel{s-s}{START-TO-START}
\setganttlinklabel{f-s}{FINISH-TO-START}
\setganttlinklabel{f-f}{FINISH-TO-FINISH}
\sffamily
\begin{ganttchart}[
		canvas/.append style={fill=none, draw=black!5, line width=.75pt},
		hgrid style/.style={draw=black!5, line width=.75pt},
		vgrid={*1{draw=black!5, line width=.75pt}},
		%		Set the Current Week Here!!!
		today=5,
		today rule/.style={
			draw=black!64,
			dash pattern=on 3.5pt off 4.5pt,
			line width=1.5pt
		},
		today label font=\small\bfseries,
		title/.style={draw=none, fill=none},
		title label font=\bfseries\footnotesize,
		title label node/.append style={below=7pt},
		include title in canvas=false,
		bar label font=\mdseries\small\color{black!70},
		bar label node/.append style={left=2cm},
		bar/.append style={draw=none, fill=black!63},
		bar incomplete/.append style={fill=barblue},
		bar progress label font=\mdseries\footnotesize\color{black!70},
		group incomplete/.append style={fill=groupblue},
		group left shift=0,
		group right shift=0,
		group height=.5,
		group peaks tip position=0,
		group label node/.append style={left=.6cm},
		group progress label font=\bfseries\small,
		link/.style={-latex, line width=1.5pt, linkred},
		link label font=\scriptsize\bfseries,
		link label node/.append style={below left=-2pt and 0pt}
	]{1}{13}
	\gantttitle[
		title label node/.append style={below left=7pt and -3pt}
	]{WEEKS:\quad1}{1}
	\gantttitlelist{2,...,13}{1} \\
	\ganttgroup[progress=57]{Literature Review + Circuit Experimentation}{1}{10} \\
	\ganttbar[
		progress=75,
		name=MemristorDeviceModelSchematic
	]{Memristor Device Model - Schematic}{1}{6} \\
	\ganttbar[
		progress=67,
		name=MemristorDeviceModelLayout
	]{Memristor Device Model - Layout}{1}{6} \\
	\ganttbar[
		progress=67,
		name=NVRAMCellDesignSchematic
	]{NVRAM Cell Design - Schematic}{1}{6} \\
	\ganttbar[
		progress=67,
		name=NVRAMCellDesignLayout
	]{NVRAM Cell Design - Layout}{1}{7} \\
	\ganttbar[
		progress=50,
		name=NVRAMRowDesignSchematic
	]{NVRAM Row Design - Schematic}{4}{10} \\
	\ganttbar[
		progress=0,
		name=NVRAMRowDesignLayout
	]{NVRAM Row Design - Layout}{4}{10} \\
	\ganttbar[
		progress=0,
		name=NVRAMSubsystemDesignSchematic
	]{NVRAM Subsystem Design - Schematic}{4}{10} \\
	\ganttbar[
		progress=0,
		name=NVRAMSubsystemDesignLayout
	]{NVRAM Subsystem Design - Layout}{4}{10} \\ [grid]
	\ganttgroup[progress=0]{AI Engine for EDA}{4}{10} \\
	\ganttbar[progress=0]{Reinforcement Learning}{4}{5} \\
	\ganttbar[progress=0]{Evolutionary Reinforcement Learning}{6}{8} \\
	\ganttbar[progress=0]{Deep Reinforcement Learning}{9}{10} \\
	%	New line ``\\'' can be omitted only for the last entry
	\ganttbar[progress=0]{Recursive Self-Improvement}{9}{10}
	\ganttlink[link type=s-s]{MemristorDeviceModelSchematic}{MemristorDeviceModelLayout}
	\ganttlink[link type=f-s]{NVRAMCellDesignSchematic}{NVRAMCellDesignLayout}
	\ganttlink[
		link type=f-f,
		link label node/.append style=left
	]{NVRAMCellDesignLayout}{NVRAMRowDesignSchematic}
\end{ganttchart}


\end{document}
